/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {} tree defination struct
 * };
 */
class Solution {
public:
    bool isSameTree(TreeNode* p, TreeNode* q) {
        if(p==NULL && q==NULL)
        {
            return true;
        }
        if(p!=NULL && q!=NULL)
        {
            return(isSameTree(p->left,q->left) &&  isSameTree(p->right,q->right)
                  && p->val==q->val);
        }
        else {
            return false;
        }
        
        
    }
};
